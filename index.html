function main(workbook: ExcelScript.Workbook) { let table: ExcelScript.Table = workbook.getTable("OfficeForms.Table"); let dataBodyRange: ExcelScript.Range = table.getRangeBetweenHeaderAndTotal(); let lastRow: ExcelScript.Range = dataBodyRange.getLastRow(); // Calcular últimas 15 filas let lastRowNumber: number = lastRow.getRowIndex() + lastRow.getRowCount(); let firstRowToCheck: number = Math.max(0, lastRowNumber - 15); // Array para evitar procesar el mismo ID múltiples veces let valoresAProcesados: string[] = []; try { // Recorrer últimas 15 filas for (let i: number = firstRowToCheck; i < lastRowNumber; i++) { let row: ExcelScript.Range = dataBodyRange.getRow(i - firstRowToCheck); // Verificar si la columna L (URL_QR) está vacía let urlQR: string | number = row.getCell(0, 11).getValue() as string | number; if (urlQR === "" || urlQR === null || urlQR === undefined) { // Obtener valor de columna A let valorColumnaA: string = row.getCell(0, 0).getValue().toString(); // Evitar procesar el mismo ID múltiples veces if (valoresAProcesados.indexOf(valorColumnaA) !== -1) { continue; } valoresAProcesados.push(valorColumnaA); // Buscar TODOS los registros con el mismo ID let registrosArray: (string | number)[][] = []; let totalRows: number = dataBodyRange.getRowCount(); for (let j: number = 0; j < totalRows; j++) { let currentRow: ExcelScript.Range = dataBodyRange.getRow(j); let currentValorA: string = currentRow.getCell(0, 0).getValue().toString(); if (currentValorA === valorColumnaA) { let np: string | number = currentRow.getCell(0, 5).getValue() as string | number; let dc: string | number = currentRow.getCell(0, 6).getValue() as string | number; let bm: string | number = currentRow.getCell(0, 7).getValue() as string | number; let bc: string | number = currentRow.getCell(0, 8).getValue() as string | number; let fb: string | number = currentRow.getCell(0, 9).getValue() as string | number; let fe: string | number = currentRow.getCell(0, 10).getValue() as string | number; let valores: (string | number)[] = [np, dc, bm, bc, fb, fe]; registrosArray.push(valores); } } // Generar ID único let timestamp: number = new Date().getTime(); let random: number = Math.floor(Math.random() * 1000); let idUnico: string = `${valorColumnaA}_${timestamp.toString().slice(-6)}${random}`; // Crear estructura compacta let datosComprimidos: object = { i: idUnico, r: valorColumnaA, c: registrosArray.length, f: ["np", "dc", "bm", "bc", "fb", "fe"], d: registrosArray }; let datosJSON: string = JSON.stringify(datosComprimidos); let datosEncoded: string = encodeURIComponent(datosJSON); let urlVisual: string = `https://rayjuans06.github.io/qr-viewer?i=${idUnico}&d=${datosEncoded}`; // Generar QR más grande let qrUrl: string = `https://api.qrserver.com/v1/create-qr-code/?size=800x800&data=${encodeURIComponent(urlVisual)}`; // Actualizar TODAS las filas con ese ID for (let j: number = 0; j < totalRows; j++) { let currentRow: ExcelScript.Range = dataBodyRange.getRow(j); let currentValorA: string = currentRow.getCell(0, 0).getValue().toString(); if (currentValorA === valorColumnaA) { currentRow.getCell(0, 11).setValue(qrUrl); currentRow.getCell(0, 12).setValue(datosJSON); currentRow.getCell(0, 13).setValue(idUnico); currentRow.getCell(0, 14).setValue(urlVisual); } } } } } catch (error) { let ws: ExcelScript.Worksheet = workbook.getActiveWorksheet(); ws.getCell(0, 0).setValue("ERROR: " + error); } }
